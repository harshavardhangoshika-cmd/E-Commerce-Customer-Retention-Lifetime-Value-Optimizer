--- TABLE: stg_customer_transactions ---
CREATE TABLE stg_customer_transactions (
	customer_id BIGINT, 
	age BIGINT, 
	gender TEXT, 
	country TEXT, 
	annual_income BIGINT, 
	spending_score BIGINT, 
	num_purchases BIGINT, 
	avg_purchase_value FLOAT, 
	membership_years BIGINT, 
	website_visits_per_month BIGINT, 
	cart_abandon_rate FLOAT, 
	churned BIGINT, 
	feedback_text TEXT, 
	last_purchase_date TEXT
);

--- VIEW: customer_summary ---
CREATE VIEW customer_summary AS
WITH dataset_constants AS (
    -- Establish a dynamic baseline reference point (Max Date + 1 Day)
    SELECT DATE(MAX(last_purchase_date), '+1 day') AS ref_date
    FROM stg_customer_transactions
)
SELECT 
    t.customer_id,
    t.age,
    t.gender,
    t.country,
    t.annual_income,
    t.spending_score,
    t.membership_years,
    t.website_visits_per_month,
    t.cart_abandon_rate,
    t.churned,
    t.feedback_text,
    
    -- --- CORE COHORT DATES ---
    MAX(t.last_purchase_date) AS latest_purchase_date,
    
    -- --- CORE RFM METRICS ---
    -- 1. RECENCY: Calculated using Julian Day differences
    CAST(julianday(c.ref_date) - julianday(MAX(t.last_purchase_date)) AS INT) AS recency_days,
    
    -- 2. FREQUENCY: Total orders recorded
    t.num_purchases AS total_orders,
    
    -- 3. MONETARY: Total lifetime financial commitment
    (t.num_purchases * t.avg_purchase_value) AS total_lifetime_spend,
    
    -- --- DERIVED FEATURE ENGINEERING ---
    CASE 
        WHEN t.num_purchases > 0 THEN (t.num_purchases * t.avg_purchase_value) / t.num_purchases
        ELSE 0 
    END AS calculated_avg_order_value

FROM 
    stg_customer_transactions t
CROSS JOIN 
    dataset_constants c
GROUP BY 
    t.customer_id;
